cmake_minimum_required(VERSION 3.16)

project(lab7 VERSION 1.0 LANGUAGES CXX)

# включаем автоматику qt
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

find_package(Qt6 REQUIRED COMPONENTS Widgets)

set(SOURCES
    main.cc
    ui/src/MainWindow.cc
    ui/src/FiguresWidget.cc
    core/src/Rect.cc
    core/src/Triag.cc
    core/src/Ellipse.cc
)

set(HEADERS
    ui/include/MainWindow.h
    ui/include/FiguresWidget.h
    core/include/Figure.h
    core/include/Rect.h
    core/include/Triag.h
    core/include/Ellipse.h
)

add_executable(${PROJECT_NAME}
    ${SOURCES}
    ${HEADERS}
)

target_include_directories(${PROJECT_NAME} PRIVATE
    ${PROJECT_SOURCE_DIR}/ui/include
    ${PROJECT_SOURCE_DIR}/core/include
)

target_link_libraries(${PROJECT_NAME} PRIVATE Qt6::Widgets)
